def daily_co2
    array = []
    food_gr.each { |val| array.push(val/100) }
    food.tail.value.list_amount_environmental_impact(food, array)
end

def index_GEI
    if daily_co2 < 800
        return 1
    elsif (800..1200) === daily_co2
        return 2
    else
        return 3
    end
end
